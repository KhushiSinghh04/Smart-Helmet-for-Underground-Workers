#include <DHT.h>

// Pin Definitions
#define VIBRATION_PIN 7
#define IR_PIN 9       
#define BUZZER_PIN 6    
#define GAS_PIN A1
#define DHT_PIN 10     

// Sensor Definitions
#define DHT_TYPE DHT11

// Object Instantiation
DHT dht(DHT_PIN, DHT_TYPE);

// Thresholds
const int GAS_THRESHOLD = 300;
int buzzerPin = 6;

void setup() {
  Serial.begin(9600);
  
  pinMode(VIBRATION_PIN, INPUT);
  pinMode(IR_PIN, INPUT);
  pinMode(BUZZER_PIN, OUTPUT);
  pinMode(GAS_PIN, INPUT);
  digitalWrite(buzzerPin, LOW);
  
  dht.begin();
}

void loop() {
  // Read sensor values
  int vibration = digitalRead(VIBRATION_PIN);  // digitalRead for vibration
  int ir = digitalRead(IR_PIN);
  float temperature = dht.readTemperature();
  float humidity = dht.readHumidity();
  delay(2000);
  int gas = analogRead(GAS_PIN);
  
  // Display sensor data on Serial Monitor
  Serial.print("Temperature: ");
  Serial.print(temperature);
  Serial.print(" C, Humidity: ");
  Serial.print(humidity);
  Serial.println(" %");
  
  Serial.print("Gas Level: ");
  Serial.print(gas);
  Serial.print(", Vibration: ");
  Serial.println(vibration);

  // Check for alerts and display on Serial Monitor
  if (vibration == HIGH) {  // HIGH means vibration detected
    alert("High Vibration");
  }
  
  if (ir == LOW) {  // LOW means object detected
    alert("Object Detected");
  }
  
  if (gas > GAS_THRESHOLD) {
    alert("Gas Detected");
  }
  
  delay(5000);  // Wait 5 seconds before next reading
}

void alert(String message) {
  tone(BUZZER_PIN, 1000);  // start the buzzer
  Serial.print("ALERT: ");
  Serial.println(message);
  delay(1000); // buzzer will beep for 1 sec
  noTone(BUZZER_PIN); // buzzer will stop
}
